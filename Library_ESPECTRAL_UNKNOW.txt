local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local RunService = game:GetService("RunService")
local CoreGui = game:GetService("CoreGui")

local library = {}

local Theme = {
    Background = Color3.fromRGB(15, 10, 22),
    Panel      = Color3.fromRGB(22, 15, 32),
    Accent     = Color3.fromRGB(150, 60, 230),
    AccentDark = Color3.fromRGB(80, 35, 130),
    Stroke     = Color3.fromRGB(45, 30, 65),
    Text       = Color3.fromRGB(235, 230, 245),
    SubText    = Color3.fromRGB(150, 140, 165),
}

local function Create(class, props)
    local obj = Instance.new(class)
    for i, v in pairs(props) do
        obj[i] = v
    end
    return obj
end

local function Tween(obj, props, time)
    TweenService:Create(obj, TweenInfo.new(time or 0.2, Enum.EasingStyle.Quad), props):Play()
end

local function Corner(parent, radius)
    Create("UICorner", {CornerRadius = UDim.new(0, radius or 6), Parent = parent})
end

local function Stroke(parent, color, thickness)
    Create("UIStroke", {Color = color or Theme.Stroke, Thickness = thickness or 1, Parent = parent})
end

-- destroy old instance
do
    local old = CoreGui:FindFirstChild("ESPECTRAL_UNKNOW")
    if old then old:Destroy() end
end

function library:CreateWindow(config)
    config = config or {}
    local title = config.Title or "ESPECTRAL UNKNOW"
    local subtitle = config.SubTitle or "v1.0"

    local Gui = Create("ScreenGui", {
        Name = "ESPECTRAL_UNKNOW",
        Parent = CoreGui,
        ResetOnSpawn = false,
        ZIndexBehavior = Enum.ZIndexBehavior.Sibling,
    })

    local Main = Create("Frame", {
        Name = "Main",
        Parent = Gui,
        BackgroundColor3 = Theme.Background,
        Position = UDim2.new(0.5, -320, 0.5, -190),
        Size = UDim2.new(0, 640, 0, 380),
        Active = true,
    })
    Corner(Main, 10)
    Stroke(Main, Theme.AccentDark, 1)

    -- Top bar
    local TopBar = Create("Frame", {
        Name = "TopBar",
        Parent = Main,
        BackgroundColor3 = Theme.Panel,
        Size = UDim2.new(1, 0, 0, 46),
    })
    Corner(TopBar, 10)

    Create("Frame", {
        Parent = TopBar,
        BackgroundColor3 = Theme.Panel,
        Position = UDim2.new(0, 0, 1, -10),
        Size = UDim2.new(1, 0, 0, 10),
        BorderSizePixel = 0,
    })

    Create("Frame", {
        Parent = TopBar,
        BackgroundColor3 = Theme.Accent,
        Position = UDim2.new(0, 0, 1, -2),
        Size = UDim2.new(1, 0, 0, 2),
    })

    Create("TextLabel", {
        Parent = TopBar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 16, 0, 6),
        Size = UDim2.new(0, 300, 0, 22),
        Font = Enum.Font.GothamBold,
        Text = title,
        TextColor3 = Theme.Text,
        TextSize = 18,
        TextXAlignment = Enum.TextXAlignment.Left,
    })

    Create("TextLabel", {
        Parent = TopBar,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 16, 0, 26),
        Size = UDim2.new(0, 300, 0, 16),
        Font = Enum.Font.Gotham,
        Text = subtitle,
        TextColor3 = Theme.Accent,
        TextSize = 12,
        TextXAlignment = Enum.TextXAlignment.Left,
    })

    -- Close / minimize
    local CloseBtn = Create("TextButton", {
        Parent = TopBar,
        BackgroundColor3 = Theme.AccentDark,
        Position = UDim2.new(1, -38, 0, 8),
        Size = UDim2.new(0, 30, 0, 30),
        Font = Enum.Font.GothamBold,
        Text = "X",
        TextColor3 = Theme.Text,
        TextSize = 14,
    })
    Corner(CloseBtn, 6)

    -- Tab bar (RIGHT SIDE)
    local TabBar = Create("Frame", {
        Name = "TabBar",
        Parent = Main,
        BackgroundColor3 = Theme.Panel,
        Position = UDim2.new(1, -110, 0, 56),
        Size = UDim2.new(0, 100, 1, -66),
    })
    Corner(TabBar, 8)
    Stroke(TabBar, Theme.Stroke, 1)

    local TabList = Create("UIListLayout", {
        Parent = TabBar,
        Padding = UDim.new(0, 6),
        SortOrder = Enum.SortOrder.LayoutOrder,
    })
    Create("UIPadding", {
        Parent = TabBar,
        PaddingTop = UDim.new(0, 8),
        PaddingLeft = UDim.new(0, 6),
        PaddingRight = UDim.new(0, 6),
    })

    -- Content area (LEFT of tab bar)
    local Content = Create("Frame", {
        Name = "Content",
        Parent = Main,
        BackgroundTransparency = 1,
        Position = UDim2.new(0, 10, 0, 56),
        Size = UDim2.new(1, -130, 1, -66),
    })

    -- dragging
    do
        local dragging, dragStart, startPos
        TopBar.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 then
                dragging = true
                dragStart = input.Position
                startPos = Main.Position
                input.Changed:Connect(function()
                    if input.UserInputState == Enum.UserInputState.End then
                        dragging = false
                    end
                end)
            end
        end)
        UserInputService.InputChanged:Connect(function(input)
            if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                local delta = input.Position - dragStart
                Main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            end
        end)
    end

    CloseBtn.MouseButton1Click:Connect(function()
        Gui.Enabled = not Gui.Enabled
    end)

    -- toggle keybind
    UserInputService.InputBegan:Connect(function(input, gpe)
        if gpe then return end
        if input.KeyCode == (config.ToggleKey or Enum.KeyCode.RightShift) then
            Gui.Enabled = not Gui.Enabled
        end
    end)

    local window = {}
    local firstTab = true

    function window:AddTab(name, icon)
        local tabBtn = Create("TextButton", {
            Parent = TabBar,
            BackgroundColor3 = Theme.AccentDark,
            BackgroundTransparency = 1,
            Size = UDim2.new(1, 0, 0, 38),
            Font = Enum.Font.GothamSemibold,
            Text = (icon and (icon .. "  ") or "") .. name,
            TextColor3 = Theme.SubText,
            TextSize = 13,
            AutoButtonColor = false,
        })
        Corner(tabBtn, 6)

        local page = Create("ScrollingFrame", {
            Parent = Content,
            BackgroundTransparency = 1,
            Size = UDim2.new(1, 0, 1, 0),
            Visible = false,
            ScrollBarThickness = 3,
            ScrollBarImageColor3 = Theme.Accent,
            CanvasSize = UDim2.new(0, 0, 0, 0),
            AutomaticCanvasSize = Enum.AutomaticSize.Y,
        })
        local pageList = Create("UIListLayout", {
            Parent = page,
            Padding = UDim.new(0, 8),
            SortOrder = Enum.SortOrder.LayoutOrder,
        })
        Create("UIPadding", {
            Parent = page,
            PaddingTop = UDim.new(0, 4),
            PaddingRight = UDim.new(0, 6),
        })

        local function select()
            for _, p in pairs(Content:GetChildren()) do
                if p:IsA("ScrollingFrame") then p.Visible = false end
            end
            for _, b in pairs(TabBar:GetChildren()) do
                if b:IsA("TextButton") then
                    Tween(b, {BackgroundTransparency = 1}, 0.15)
                    TweenService:Create(b, TweenInfo.new(0.15), {TextColor3 = Theme.SubText}):Play()
                end
            end
            page.Visible = true
            Tween(tabBtn, {BackgroundTransparency = 0}, 0.15)
            TweenService:Create(tabBtn, TweenInfo.new(0.15), {TextColor3 = Theme.Text}):Play()
        end

        tabBtn.MouseButton1Click:Connect(select)

        if firstTab then
            firstTab = false
            select()
        end

        local tab = {}

        function tab:AddLabel(text)
            local lbl = Create("TextLabel", {
                Parent = page,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, 20),
                Font = Enum.Font.Gotham,
                Text = text,
                TextColor3 = Theme.SubText,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
            })
            return lbl
        end

        function tab:AddButton(text, callback)
            callback = callback or function() end
            local btn = Create("TextButton", {
                Parent = page,
                BackgroundColor3 = Theme.Panel,
                Size = UDim2.new(1, 0, 0, 36),
                Font = Enum.Font.GothamSemibold,
                Text = text,
                TextColor3 = Theme.Text,
                TextSize = 13,
                AutoButtonColor = false,
            })
            Corner(btn, 6)
            Stroke(btn, Theme.Stroke, 1)

            btn.MouseEnter:Connect(function() Tween(btn, {BackgroundColor3 = Theme.AccentDark}, 0.15) end)
            btn.MouseLeave:Connect(function() Tween(btn, {BackgroundColor3 = Theme.Panel}, 0.15) end)
            btn.MouseButton1Click:Connect(function() callback() end)

            return btn
        end

        function tab:AddToggle(text, default, callback)
            callback = callback or function() end
            local state = default or false

            local holder = Create("Frame", {
                Parent = page,
                BackgroundColor3 = Theme.Panel,
                Size = UDim2.new(1, 0, 0, 36),
            })
            Corner(holder, 6)
            Stroke(holder, Theme.Stroke, 1)

            Create("TextLabel", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 12, 0, 0),
                Size = UDim2.new(1, -60, 1, 0),
                Font = Enum.Font.Gotham,
                Text = text,
                TextColor3 = Theme.Text,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
            })

            local switchBG = Create("Frame", {
                Parent = holder,
                BackgroundColor3 = state and Theme.Accent or Theme.AccentDark,
                Position = UDim2.new(1, -46, 0.5, -10),
                Size = UDim2.new(0, 34, 0, 20),
            })
            Corner(switchBG, 10)

            local knob = Create("Frame", {
                Parent = switchBG,
                BackgroundColor3 = Theme.Text,
                Position = state and UDim2.new(1, -18, 0.5, -8) or UDim2.new(0, 2, 0.5, -8),
                Size = UDim2.new(0, 16, 0, 16),
            })
            Corner(knob, 8)

            local click = Create("TextButton", {
                Parent = holder,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 1, 0),
                Text = "",
            })

            click.MouseButton1Click:Connect(function()
                state = not state
                Tween(switchBG, {BackgroundColor3 = state and Theme.Accent or Theme.AccentDark}, 0.15)
                Tween(knob, {Position = state and UDim2.new(1, -18, 0.5, -8) or UDim2.new(0, 2, 0.5, -8)}, 0.15)
                callback(state)
            end)

            return {
                Set = function(_, v)
                    state = v
                    Tween(switchBG, {BackgroundColor3 = state and Theme.Accent or Theme.AccentDark}, 0.15)
                    Tween(knob, {Position = state and UDim2.new(1, -18, 0.5, -8) or UDim2.new(0, 2, 0.5, -8)}, 0.15)
                    callback(state)
                end
            }
        end

        function tab:AddSlider(text, min, max, default, callback)
            callback = callback or function() end
            min, max = min or 0, max or 100
            local value = math.clamp(default or min, min, max)

            local holder = Create("Frame", {
                Parent = page,
                BackgroundColor3 = Theme.Panel,
                Size = UDim2.new(1, 0, 0, 50),
            })
            Corner(holder, 6)
            Stroke(holder, Theme.Stroke, 1)

            Create("TextLabel", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 12, 0, 6),
                Size = UDim2.new(1, -70, 0, 18),
                Font = Enum.Font.Gotham,
                Text = text,
                TextColor3 = Theme.Text,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
            })

            local valLabel = Create("TextLabel", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(1, -60, 0, 6),
                Size = UDim2.new(0, 50, 0, 18),
                Font = Enum.Font.GothamBold,
                Text = tostring(value),
                TextColor3 = Theme.Accent,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Right,
            })

            local barBG = Create("Frame", {
                Parent = holder,
                BackgroundColor3 = Theme.AccentDark,
                Position = UDim2.new(0, 12, 1, -14),
                Size = UDim2.new(1, -24, 0, 6),
            })
            Corner(barBG, 3)

            local function ratio(v) return (v - min) / (max - min) end

            local fill = Create("Frame", {
                Parent = barBG,
                BackgroundColor3 = Theme.Accent,
                Size = UDim2.new(ratio(value), 0, 1, 0),
            })
            Corner(fill, 3)

            local dragging = false
            local function update(input)
                local rel = math.clamp((input.Position.X - barBG.AbsolutePosition.X) / barBG.AbsoluteSize.X, 0, 1)
                value = math.floor(min + (max - min) * rel)
                fill.Size = UDim2.new(rel, 0, 1, 0)
                valLabel.Text = tostring(value)
                callback(value)
            end

            barBG.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 then
                    dragging = true
                    update(input)
                end
            end)
            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end
            end)
            UserInputService.InputChanged:Connect(function(input)
                if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
                    update(input)
                end
            end)

            return {
                Set = function(_, v)
                    value = math.clamp(v, min, max)
                    fill.Size = UDim2.new(ratio(value), 0, 1, 0)
                    valLabel.Text = tostring(value)
                    callback(value)
                end
            }
        end

        function tab:AddDropdown(text, options, default, callback)
            callback = callback or function() end
            options = options or {}
            local open = false
            local selected = default or options[1] or "..."

            local holder = Create("Frame", {
                Parent = page,
                BackgroundColor3 = Theme.Panel,
                Size = UDim2.new(1, 0, 0, 36),
                ClipsDescendants = true,
                ZIndex = 2,
            })
            Corner(holder, 6)
            Stroke(holder, Theme.Stroke, 1)

            local main = Create("TextButton", {
                Parent = holder,
                BackgroundTransparency = 1,
                Size = UDim2.new(1, 0, 0, 36),
                Font = Enum.Font.Gotham,
                Text = "",
                TextColor3 = Theme.Text,
                ZIndex = 3,
            })

            Create("TextLabel", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 12, 0, 0),
                Size = UDim2.new(1, -90, 0, 36),
                Font = Enum.Font.Gotham,
                Text = text,
                TextColor3 = Theme.Text,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
                ZIndex = 3,
            })

            local selLabel = Create("TextLabel", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(1, -90, 0, 0),
                Size = UDim2.new(0, 80, 0, 36),
                Font = Enum.Font.GothamBold,
                Text = tostring(selected),
                TextColor3 = Theme.Accent,
                TextSize = 12,
                TextXAlignment = Enum.TextXAlignment.Right,
                ZIndex = 3,
            })

            local list = Create("Frame", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 0, 0, 40),
                Size = UDim2.new(1, 0, 0, #options * 26),
                ZIndex = 3,
            })
            Create("UIListLayout", {Parent = list, SortOrder = Enum.SortOrder.LayoutOrder, Padding = UDim.new(0, 2)})

            for _, opt in ipairs(options) do
                local optBtn = Create("TextButton", {
                    Parent = list,
                    BackgroundColor3 = Theme.AccentDark,
                    BackgroundTransparency = 0.5,
                    Size = UDim2.new(1, 0, 0, 24),
                    Font = Enum.Font.Gotham,
                    Text = "  " .. tostring(opt),
                    TextColor3 = Theme.Text,
                    TextXAlignment = Enum.TextXAlignment.Left,
                    TextSize = 12,
                    AutoButtonColor = false,
                })
                Corner(optBtn, 4)
                optBtn.MouseButton1Click:Connect(function()
                    selected = opt
                    selLabel.Text = tostring(opt)
                    open = false
                    Tween(holder, {Size = UDim2.new(1, 0, 0, 36)}, 0.15)
                    callback(opt)
                end)
            end

            main.MouseButton1Click:Connect(function()
                open = not open
                Tween(holder, {Size = open and UDim2.new(1, 0, 0, 40 + #options * 26) or UDim2.new(1, 0, 0, 36)}, 0.15)
            end)

            return {
                Set = function(_, v)
                    selected = v
                    selLabel.Text = tostring(v)
                    callback(v)
                end
            }
        end

        function tab:AddTextbox(text, placeholder, callback)
            callback = callback or function() end

            local holder = Create("Frame", {
                Parent = page,
                BackgroundColor3 = Theme.Panel,
                Size = UDim2.new(1, 0, 0, 36),
            })
            Corner(holder, 6)
            Stroke(holder, Theme.Stroke, 1)

            Create("TextLabel", {
                Parent = holder,
                BackgroundTransparency = 1,
                Position = UDim2.new(0, 12, 0, 0),
                Size = UDim2.new(0.4, 0, 1, 0),
                Font = Enum.Font.Gotham,
                Text = text,
                TextColor3 = Theme.Text,
                TextSize = 13,
                TextXAlignment = Enum.TextXAlignment.Left,
            })

            local box = Create("TextBox", {
                Parent = holder,
                BackgroundColor3 = Theme.AccentDark,
                Position = UDim2.new(0.4, 0, 0.15, 0),
                Size = UDim2.new(0.6, -12, 0.7, 0),
                Font = Enum.Font.Gotham,
                PlaceholderText = placeholder or "...",
                Text = "",
                TextColor3 = Theme.Text,
                TextSize = 12,
                ClearTextOnFocus = false,
            })
            Corner(box, 4)

            box.FocusLost:Connect(function(enter)
                callback(box.Text, enter)
            end)

            return box
        end

        return tab
    end

    return window
end

return library
